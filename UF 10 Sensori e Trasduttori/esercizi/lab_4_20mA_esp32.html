C++ · Arduino / ESP32es1_ricevitore.ino
/*
 * ES1 – Ricevitore 4-20 mA su ESP32
 * Il potenziometro simula il trasmettitore industriale.
 * R_shunt = 150 Ω converte la corrente in tensione per l’ADC.
 */

#include <Arduino.h>

constexpr int    PIN_ADC      = 34;
constexpr int    ADC_MAX      = 4095;
constexpr float  VREF         = 3.3f;
constexpr float  R_SHUNT      = 150.0f;   // Ω
constexpr float  I_MIN        = 4.0f;     // mA – live zero
constexpr float  I_MAX        = 20.0f;    // mA – fondo scala
constexpr float  TEMP_MIN     = 0.0f;
constexpr float  TEMP_MAX     = 100.0f;
constexpr int    N_SAMPLES    = 16;       // oversampling

void setup() {
  Serial.begin(115200);
  analogReadResolution(12);
  analogSetAttenuation(ADC_ATTEN_DB_11);  // range 0–3.1 V
  Serial.println("=== ES1 Ricevitore ===");
  Serial.println("raw\t\tV_sh(V)\t\tI(mA)\t\tT(C)\t\tStato");
}

void loop() {
  // 1. Oversampling
  long acc = 0;
  for (int i = 0; i < N_SAMPLES; ++i) acc += analogRead(PIN_ADC);
  const int   raw     = (int)(acc / N_SAMPLES);

  // 2. ADC → tensione
  const float v_sh    = (float)raw / ADC_MAX * VREF;

  // 3. Tensione → corrente
  const float i_loop  = (v_sh / R_SHUNT) * 1000.0f;

  // 4. Corrente → grandezza fisica
  const float frac    = constrain((i_loop - I_MIN)/(I_MAX - I_MIN), 0.f, 1.f);
  const float temp    = TEMP_MIN + (TEMP_MAX - TEMP_MIN) * frac;

  // 5. Diagnostica
  const char* stato =
    (i_loop < 2.0f) ? "LOOP APERTO" :
    (i_loop < I_MIN) ? "SOTTO SCALA" :
    (i_loop > I_MAX) ? "SATURAZIONE" : "OK";

  Serial.print(raw);       Serial.print("\t\t");
  Serial.print(v_sh,  4);  Serial.print("\t\t");
  Serial.print(i_loop,3);  Serial.print("\t\t");
  Serial.print(temp,  2);  Serial.print("\t\t");
  Serial.println(stato);
  delay(500);
}

C++ · ESP32es2_trasmettitore.ino
/*
 * ES2 – Trasmettitore + Ricevitore sullo stesso ESP32
 * GPIO25 = DAC output  (simula il trasmettitore)
 * GPIO34 = ADC input   (simula il ricevitore con R_shunt)
 * Scansione automatica 0→100°C a passi di 10°C
 */

#include <Arduino.h>

constexpr int   DAC_PIN  = 25;
constexpr int   ADC_PIN  = 34;
constexpr float VREF     = 3.3f;
constexpr float R_SHUNT  = 150.0f;
constexpr float I_MIN    = 4.0f;
constexpr float I_MAX    = 20.0f;

// Converte temperatura [0-100 C] in valore DAC [0-255]
uint8_t tempToDAC(float temp) {
  const float i   = I_MIN + (I_MAX - I_MIN) * (temp / 100.0f);
  const float v   = (i / 1000.0f) * R_SHUNT;
  return (uint8_t) constrain(v / VREF * 255.0f, 0, 255);
}

void setup() {
  Serial.begin(115200);
  analogReadResolution(12);
  analogSetAttenuation(ADC_ATTEN_DB_11);
  Serial.println("T_TX\tDAC\tV_TX(V)\tI_TX(mA)\tI_RX(mA)\tErrore");
}

void loop() {
  for (float t = 0.0f; t <= 100.0f; t += 10.0f) {

    // — TX —
    const uint8_t dv  = tempToDAC(t);
    dacWrite(DAC_PIN, dv);
    delay(80);  // stabilizzazione
    const float v_tx  = dv / 255.0f * VREF;
    const float i_tx  = (v_tx / R_SHUNT) * 1000.0f;

    // — RX —
    long acc = 0;
    for(int i=0; i<16; ++i) acc += analogRead(ADC_PIN);
    const float v_rx  = (float)(acc/16) / 4095.0f * VREF;
    const float i_rx  = (v_rx / R_SHUNT) * 1000.0f;

    Serial.print(t,    1);  Serial.print("\t");
    Serial.print(dv);       Serial.print("\t");
    Serial.print(v_tx, 4);  Serial.print("\t");
    Serial.print(i_tx, 3);  Serial.print("\t\t");
    Serial.print(i_rx, 3);  Serial.print("\t\t");
    Serial.println(i_rx - i_tx, 3);
  }
  Serial.println("---"); delay(2000);
}

#include <Arduino.h>
constexpr int   DAC  = 25;
constexpr float RSH  = 150.f;
constexpr float VR   = 3.3f;

uint8_t toDAC(float t){
  float i=4.f+16.f*(t/100.f);
  return (uint8_t)(i/1000.f*RSH/VR*255.f);
}

void setup(){Serial.begin(115200);}

void loop(){
  for(float t=0;t<=100;t+=5){
    dacWrite(DAC,toDAC(t));
    Serial.print("TX=");
    Serial.println(t);
    delay(500);
  }
  for(float t=100;t>=0;t-=5){
    dacWrite(DAC,toDAC(t));
    Serial.print("TX=");
    Serial.println(t);

  }
}

C++ · ESP32es4_allarmi.ino
/*
 * ES4 – Sistema di allarme 4-20 mA con isteresi
 * Soglie: T_LOW=20°C, T_HIGH=80°C, GUASTO <2mA
 * Isteresi: 2°C (banda morta) per evitare oscillazioni
 */

#include <Arduino.h>

constexpr int   PIN_ADC   = 34;
constexpr int   LED_OK    = 2;   // verde
constexpr int   LED_LOW   = 4;   // giallo
constexpr int   LED_HIGH  = 5;   // rosso
constexpr int   LED_FAULT = 18;  // blu – guasto cavo

constexpr float VREF      = 3.3f;
constexpr float R_SHUNT   = 150.0f;
constexpr float I_MIN     = 4.0f;
constexpr float I_MAX     = 20.0f;
constexpr float TEMP_MIN  = 0.0f;
constexpr float TEMP_MAX  = 100.0f;
constexpr float T_LOW     = 20.0f;    // soglia bassa
constexpr float T_HIGH    = 80.0f;    // soglia alta
constexpr float ISTERESI  = 2.0f;     // °C banda morta
constexpr float I_FAULT   = 2.0f;     // mA soglia guasto

enum class Stato { OK, BASSA, ALTA, GUASTO };
Stato stato_prec = Stato::OK;

void setLEDs(Stato s) {
  digitalWrite(LED_OK,    s==Stato::OK     ? HIGH:LOW);
  digitalWrite(LED_LOW,   s==Stato::BASSA  ? HIGH:LOW);
  digitalWrite(LED_HIGH,  s==Stato::ALTA   ? HIGH:LOW);
  digitalWrite(LED_FAULT, s==Stato::GUASTO ? HIGH:LOW);
}

Stato valuta(float iloop, float t, Stato prec) {
  if (iloop < I_FAULT) return Stato::GUASTO;
  // isteresi soglia alta
  if (prec==Stato::ALTA  && t > T_HIGH-ISTERESI) return Stato::ALTA;
  if (prec!=Stato::ALTA  && t > T_HIGH+ISTERESI) return Stato::ALTA;
  // isteresi soglia bassa
  if (prec==Stato::BASSA && t < T_LOW+ISTERESI)  return Stato::BASSA;
  if (prec!=Stato::BASSA && t < T_LOW-ISTERESI)  return Stato::BASSA;
  return Stato::OK;
}

void setup() {
  Serial.begin(115200);
  analogReadResolution(12);
  analogSetAttenuation(ADC_ATTEN_DB_11);
  for (int p : {LED_OK, LED_LOW, LED_HIGH, LED_FAULT})
    pinMode(p, OUTPUT);
  Serial.println("I(mA)\tT(C)\tStato");
}

void loop() {
  long acc=0;
  for(int i=0;i<16;++i) acc+=analogRead(PIN_ADC);
  const float v  = (float)(acc/16)/4095.f*VREF;
  const float il = v/R_SHUNT*1000.f;
  const float t  = TEMP_MIN +
    (TEMP_MAX-TEMP_MIN)*constrain((il-I_MIN)/(I_MAX-I_MIN),0.f,1.f);

  const Stato s = valuta(il, t, stato_prec);
  stato_prec = s;
  setLEDs(s);

  const char* n =
    s==Stato::OK    ?"OK"          :
    s==Stato::BASSA ?"T BASSA"    :
    s==Stato::ALTA  ?"T ALTA"     :"GUASTO CAVO";

  Serial.print(il,3); Serial.print("\t");
  Serial.print(t, 2); Serial.print("\t");
  Serial.println(n);
  delay(300);
}

Esercizio 5 · Calibrazione
Calibrazione a due punti — correzione non-linearità ADC
Obiettivi
▸Misurare l’errore di offset e guadagno dell’ADC ESP32
▸Acquisire due punti di calibrazione (4 mA e 20 mA)
▸Applicare la correzione lineare: I_cal = m × raw + q
▸Confrontare l’errore prima e dopo la calibrazione
💡 L’ADC interno dell’ESP32 ha una non-linearità integrale (INL) tipica di ±10 LSB e un offset variabile da chip a chip. La calibrazione a due punti corregge l’errore di guadagno e di offset con una semplice regressione lineare su due valori noti.
Procedura di calibrazione — passi
Punto LOW (4 mA): regola il potenziometro finché il Serial Monitor mostra I_loop ≈ 4.00 mA (calcolato da V_shunt). Leggi e annota raw_low. Il valore atteso è V_shunt = 4 mA × 150 Ω = 0.600 V → raw_teorico = 0.600/3.3 × 4095 ≈ 744.
Punto HIGH (20 mA): porta il pot al massimo (I_loop ≈ 20 mA). Annota raw_high. Atteso: V = 3.0 V → raw ≈ 3723.
Calcola coefficienti: m = (20.0 - 4.0) / (raw_high - raw_low) e q = 4.0 - m × raw_low.
Inserisci i valori nelle costanti CAL_M e CAL_Q nel codice e riavvia la simulazione.
Confronta la colonna I_raw (non calibrata) con I_cal (calibrata). Calcola l’errore in entrambi i casi.
C++ · ESP32es5_calibrazione.ino
/*
 * ES5 – Calibrazione a due punti dell’ADC ESP32
 * I_loop = m * raw + q
 * m e q vengono determinati misurando due punti noti:
 * LOW = 4 mA (live zero), HIGH = 20 mA (fondo scala)
 */

#include <Arduino.h>

constexpr int   PIN_ADC  = 34;
constexpr float VREF     = 3.3f;
constexpr float R_SHUNT  = 150.0f;

// ── Punti di calibrazione (da misurare e inserire) ────────
// Fase 1: lascia CAL_M=1.0 e CAL_Q=0.0, misura raw_low e raw_high
// Fase 2: calcola m e q e sostituisci qui sotto
constexpr float RAW_LOW  = 744.0f;   // raw a 4 mA  (da misurare)
constexpr float RAW_HIGH = 3723.0f;  // raw a 20 mA (da misurare)
constexpr float I_LOW    = 4.0f;
constexpr float I_HIGH   = 20.0f;

// Coefficienti calcolati dalla calibrazione
constexpr float CAL_M = (I_HIGH - I_LOW) / (RAW_HIGH - RAW_LOW);
constexpr float CAL_Q = I_LOW - CAL_M * RAW_LOW;

float rawToI_uncal(int raw) {
  // Conversione senza calibrazione (formula base)
  return ((float)raw / 4095.0f * VREF / R_SHUNT) * 1000.0f;
}

float rawToI_cal(int raw) {
  // Conversione con calibrazione a due punti
  return CAL_M * (float)raw + CAL_Q;
}

void setup() {
  Serial.begin(115200);
  analogReadResolution(12);
  analogSetAttenuation(ADC_ATTEN_DB_11);
  Serial.print("CAL_M = "); Serial.println(CAL_M, 6);
  Serial.print("CAL_Q = "); Serial.println(CAL_Q, 4);
  Serial.println("raw\t\tI_raw(mA)\tI_cal(mA)\terr_raw\t\terr_cal");
}

void loop() {
  long acc = 0;
  for(int i=0; i<64; ++i) acc += analogRead(PIN_ADC);
  const int   raw    = (int)(acc / 64);
  const float i_raw  = rawToI_uncal(raw);
  const float i_cal  = rawToI_cal(raw);
  // Valore "vero" stimato dalla posizione pot (da confrontare manualmente)
  Serial.print(raw);            Serial.print("\t\t");
  Serial.print(i_raw,  4);     Serial.print("\t\t");
  Serial.print(i_cal,  4);     Serial.print("\t\t");
  Serial.print(i_raw - i_cal, 4); Serial.print("\t\t");
  Serial.println(0.0f, 4);  // err_cal = 0 per definizione sui punti di cal
  delay(400);
}

Esercizio 6 · Oversampling
Oversampling & ENOB — confronto 1, 16, 64, 256 campioni
Obiettivi
▸Misurare il rumore ADC con 1 campione e con oversampling crescente
▸Calcolare la deviazione standard del segnale per ogni N
▸Verificare la legge teorica: ENOB aumenta di 0.5 bit per ogni 4× campioni
▸Trovare il compromesso N ottimale per la velocità di campionamento richiesta
Teoria — oversampling e ENOB

Mediando N campioni indipendenti il rumore si riduce di √N. Quadruplicare N dimezza il rumore, guadagnando 1 bit di risoluzione effettiva (ENOB):

ENOB = N_bits + 0.5 × log₂(N_campioni)     [bit]

Esempio ESP32 (12-bit): con N=16 → ENOB = 12 + 0.5×4 = 14 bit effettivi. Con N=256 → ENOB = 16 bit — equivalente a un ADC esterno di precisione.

C++ · ESP32es6_oversampling.ino
/*
 * ES6 – Studio dell’oversampling sull’ADC ESP32
 * Per ogni N: acquisisce WINDOW campioni mediati, calcola
 * media e deviazione standard, stampa noise_rms in mA
 */

#include <Arduino.h>
#include <cmath>

constexpr int   PIN_ADC = 34;
constexpr float VREF    = 3.3f;
constexpr float R_SH    = 150.0f;
constexpr int   WINDOW  = 128;   // campioni per calcolo std dev

constexpr float rawToI(float r) {
  return r / 4095.0f * VREF / R_SH * 1000.0f;
}

// Acquisisce WINDOW valori mediati con N campioni ciascuno,
// ritorna {media, std_dev} della corrente in mA
struct Stats { float mean, stddev; };

Stats measure(int N) {
  float vals[WINDOW];
  for(int w=0; w<WINDOW; ++w) {
    long acc=0;
    for(int i=0; i<N; ++i) acc+=analogRead(PIN_ADC);
    vals[w] = rawToI((float)(acc/N));
  }
  // Calcolo media
  float sum=0;
  for(int i=0;i<WINDOW;++i) sum+=vals[i];
  const float mean = sum/WINDOW;
  // Calcolo deviazione standard
  float var=0;
  for(int i=0;i<WINDOW;++i) var+=(vals[i]-mean)*(vals[i]-mean);
  return {mean, sqrtf(var/WINDOW)};
}

void setup() {
  Serial.begin(115200);
  analogReadResolution(12);
  analogSetAttenuation(ADC_ATTEN_DB_11);
  Serial.println("N\tI_mean(mA)\tNoise_rms(mA)\tENOB_teorico");
}

void loop() {
  const int Nlist[] = {1, 4, 16, 64, 256};
  for (int N : Nlist) {
    const Stats s = measure(N);
    const float enob = 12.0f + 0.5f*log2f((float)N);
    Serial.print(N);            Serial.print("\t");
    Serial.print(s.mean,  4);  Serial.print("\t\t");
    Serial.print(s.stddev,5);  Serial.print("\t\t");
    Serial.println(enob,   1);
  }
  Serial.println("---"); delay(3000);
}

/*
 * ES7 – Ricevitore 4-20 mA per sensore di pressione 0-10 bar
 * Circuito identico a ES1: potenziometro + R_shunt 150 ohm + GPIO34
 * Si cambia solo il range della grandezza fisica
 */

#include <Arduino.h>

constexpr int   PIN_ADC   = 34;
constexpr float VREF      = 3.3f;
constexpr float R_SHUNT   = 150.0f;
constexpr float I_MIN     = 4.0f;
constexpr float I_MAX     = 20.0f;
constexpr int   N_SAMPLES = 16;

// ── Solo queste due costanti cambiano rispetto a ES1 ─────
constexpr float PHYS_MIN  = 0.0f;    // bar – pressione a 4 mA
constexpr float PHYS_MAX  = 10.0f;   // bar – pressione a 20 mA
constexpr char  UNIT[]    = "bar";
constexpr char  SENSOR[]  = "PT";    // Pressure Transmitter

// Soglie di allarme pressione
constexpr float P_LOW_ALARM  = 1.0f;   // bar
constexpr float P_HIGH_ALARM = 8.5f;   // bar

void setup() {
  Serial.begin(115200);
  analogReadResolution(12);
  analogSetAttenuation(ADC_ATTEN_DB_11);
  // Risoluzione in bar
  const float lsb_mA   = (I_MAX - I_MIN) / 4095.0f;
  const float lsb_bar  = lsb_mA / (I_MAX - I_MIN) * (PHYS_MAX - PHYS_MIN);
  Serial.print("Risoluzione: "); Serial.print(lsb_bar*1000,2); Serial.println(" mbar/LSB");
  Serial.println("I(mA)\t\tP(bar)\t\tStato");
}

void loop() {
  long acc=0;
  for(int i=0;i<N_SAMPLES;++i) acc+=analogRead(PIN_ADC);
  const float v     = (float)(acc/N_SAMPLES)/4095.f*VREF;
  const float i_lp  = v/R_SHUNT*1000.f;
  const float phys  = PHYS_MIN +
    (PHYS_MAX-PHYS_MIN)*constrain((i_lp-I_MIN)/(I_MAX-I_MIN),0.f,1.f);

  const char* stato =
    (i_lp  < 2.0f)        ? "GUASTO CAVO"     :
    (phys  < P_LOW_ALARM)  ? "PRESSIONE BASSA" :
    (phys  > P_HIGH_ALARM) ? "PRESSIONE ALTA"  : "OK";

  Serial.print(SENSOR); Serial.print(" ");
  Serial.print(i_lp, 3); Serial.print("\t\t");
  Serial.print(phys,  3); Serial.print(" "); Serial.print(UNIT);
  Serial.print("\t\t"); Serial.println(stato);
  delay(500);
}

/*
 * ES9 – Datalogger 4-20 mA su SD card
 * Salva su /log_NNN.csv: timestamp_ms, raw, I_mA, T_C, stato
 * Nuovo file ogni MAX_ROWS righe. LED rosso = scrittura SD.

#include <Arduino.h>
#include <SD.h>
#include <SPI.h>

constexpr int   PIN_ADC   = 34;
constexpr int   PIN_SD_CS = 5;
constexpr int   LED_SD    = 2;   // lampeggia durante scrittura
constexpr float VREF      = 3.3f;
constexpr float R_SH      = 150.0f;
constexpr float I_MIN     = 4.0f;
constexpr float I_MAX     = 20.0f;
constexpr float T_MIN     = 0.0f;
constexpr float T_MAX     = 100.0f;

int    row_count  = 0;
File   logFile;

String fileName() {
  return String("/log_") + String(file_idx, DEC) + String(".csv");
}

void openNewFile() {
  if (logFile) logFile.close();
  logFile = SD.open(fileName(), FILE_WRITE);
  if (logFile) {
    logFile.println("timestamp_ms,raw,I_mA,T_C,stato");
    Serial.print("Nuovo file: "); Serial.println(fileName());
  } else {
    Serial.println("ERRORE apertura file SD!");

  row_count = 0;
  file_idx++;
}

void setup() {
  Serial.begin(115200);
  analogReadResolution(12);
  analogSetAttenuation(ADC_ATTEN_DB_11);
  pinMode(LED_SD, OUTPUT);
  if (!SD.begin(PIN_SD_CS))
    Serial.println("SD non trovata! Verifica cablaggio.");
  openNewFile();
}

void loop() {
  static unsigned long last = 0;
  if (millis() - last < T_LOG) return;
  last = millis();

  // Lettura ADC
  long acc=0;
  for(int i=0;i<16;++i) acc+=analogRead(PIN_ADC);
  const int   raw  = (int)(acc/16);
  const float v    = (float)raw/4095.f*VREF;
  const float il   = v/R_SH*1000.f;
  const float t    = T_MIN+(T_MAX-T_MIN)*constrain((il-I_MIN)/(I_MAX-I_MIN),0.f,1.f);
  const char* s    = (il<2.f) ? "GUASTO" : (t>80.f) ? "ALTA" : (t<20.f) ? "BASSA" : "OK";

  // Scrittura CSV
  digitalWrite(LED_SD, HIGH);
  if (logFile) {
    logFile.print(millis()); logFile.print(",");
    logFile.print(raw);      logFile.print(",");
    logFile.print(il, 4);    logFile.print(",");
    logFile.print(t,  2);    logFile.print(",");
    logFile.println(s);
    logFile.flush();  // garantisce scrittura su SD
    row_count++;
    if (row_count >= MAX_ROWS) openNewFile();
  }
  digitalWrite(LED_SD, LOW);

  Serial.print(millis()); Serial.print("\t");
  Serial.print(il,3);    Serial.print("\t");
  Serial.print(t, 2);    Serial.print("\t");
  Serial.print(s);       Serial.print("\t righe: ");
  Serial.println(row_count);
}

/*
 * ES10 – 4-20 mA su MQTT via WiFi
 * Broker: test.mosquitto.org:1883
 * Publish: its/4_20mA/current, /temp, /status
 * Subscribe: its/4_20mA/cmd/t_high, /t_low
 */

#include <Arduino.h>
#include <WiFi.h>
#include <PubSubClient.h>

// ── WiFi ──────────────────────────────────────────────────
const char* SSID     = "Wokwi-GUEST";  // rete simulata Wokwi
const char* PASSWORD = "";

// ── MQTT broker ───────────────────────────────────────────
const char* BROKER   = "test.mosquitto.org";
const int   PORT     = 1883;
const char* CLIENT_ID= "esp32_4_20mA_001";

// ── Costanti fisiche ───────────────────────────────────────
constexpr int   PIN_ADC = 34;
constexpr float VREF    = 3.3f;
constexpr float R_SH    = 150.0f;
constexpr float I_MIN   = 4.0f;
constexpr float I_MAX   = 20.0f;
constexpr float T_MIN   = 0.0f;
constexpr float T_MAX   = 100.0f;

float t_high = 80.0f;   // soglia modificabile via MQTT
float t_low  = 20.0f;

WiFiClient   espClient;
PubSubClient mqtt(espClient);

// Callback: riceve messaggi dai topic sottoscritti
void mqttCallback(char* topic, byte* payload, unsigned int len) {
  char buf[16] = {};
  memcpy(buf, payload, min(len, (unsigned int)15));
  const float val = atof(buf);
  if (strcmp(topic, "its/4_20mA/cmd/t_high") == 0) {
    t_high = val;
    Serial.print("Nuova soglia alta: "); Serial.println(t_high);
  } else if (strcmp(topic, "its/4_20mA/cmd/t_low") == 0) {
    t_low = val;
    Serial.print("Nuova soglia bassa: "); Serial.println(t_low);
  }
}

void reconnect() {
  while (!mqtt.connected()) {
    Serial.print("MQTT connect...");
    if (mqtt.connect(CLIENT_ID)) {
      Serial.println(" OK");
      mqtt.subscribe("its/4_20mA/cmd/t_high");
      mqtt.subscribe("its/4_20mA/cmd/t_low");
    } else {
      Serial.print("fail rc="); Serial.print(mqtt.state()); delay(3000);
    }
  }
}

void setup() {
  Serial.begin(115200);
  analogReadResolution(12);
  analogSetAttenuation(ADC_ATTEN_DB_11);
  WiFi.begin(SSID, PASSWORD);
  Serial.print("WiFi");
  while(WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); }
  Serial.println(" connesso: " + WiFi.localIP().toString());
  mqtt.setServer(BROKER, PORT);
  mqtt.setCallback(mqttCallback);
}

void loop() {
  if (!mqtt.connected()) reconnect();
  mqtt.loop();

  static unsigned long last = 0;
  if (millis() - last < 1000) return;
  last = millis();

  long acc=0;
  for(int i=0;i<16;++i) acc+=analogRead(PIN_ADC);
  const float v  = (float)(acc/16)/4095.f*VREF;
  const float il = v/R_SH*1000.f;
  const float t  = T_MIN+(T_MAX-T_MIN)*constrain((il-I_MIN)/(I_MAX-I_MIN),0.f,1.f);
  const char* s  = (il<2.f) ? "GUASTO":(t>t_high)?"ALTA":(t<t_low)?"BASSA":"OK";

  char buf[16];
  snprintf(buf, sizeof(buf), "%.4f", il);
  mqtt.publish("its/4_20mA/current", buf);
  snprintf(buf, sizeof(buf), "%.2f", t);
  mqtt.publish("its/4_20mA/temp", buf);
  mqtt.publish("its/4_20mA/status", s);

  Serial.printf("MQTT >> I=%.3f mA  T=%.2f C  [%s]\n", il, t, s);
}
